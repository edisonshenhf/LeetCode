class Solution {
    public long minimumHealth(int[] damage, int armor) {
        long res= 0;
        int max = 0;
        int n = damage.length;
        for(int i=0;i<n;i++){
            res+=damage[i];
            max=Math.max(max,damage[i]);
        }
        if(max>=armor) res-=armor;
        else res-=max;
        return res+1;
    }
}
