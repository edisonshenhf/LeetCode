class Solution {
    public int hIndex(int[] citations) {
        int n =citations.length;
        int l = 0;
        int r = n-1;
        while(l<r){
            int mid = l+r>>1;
            if(citations[mid]>=n-mid) r=mid;
            else l=mid+1;
        }
        return Math.min(citations[r],n-r);
    }
}
