class Solution {
    public int rotatedDigits(int n) {
        int res= 0;
        for(int i=1;i<=n;i++){
            if(check(i)) res++;
        }
        return res;
    }

    boolean check(int num){
        boolean flag = false;
        while(num!=0){
            int t =num%10;
            if(t==3||t==4||t==7) return false;
            if(t==2||t==5||t==6||t==9) flag=true;
            num/=10;
        }
        return flag;
    }
}
