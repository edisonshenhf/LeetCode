class Solution {
    HashMap<Integer,Integer> map = new HashMap<>();
    public int[] anagramMappings(int[] nums1, int[] nums2) {
        int n = nums1.length;
        int[] res = new int[n];
        for(int i=0;i<n;i++){
            map.put(nums2[i],i);
        }
        for(int i=0;i<n;i++){
            res[i]=map.get(nums1[i]);
        }
        return res;
    }
}
