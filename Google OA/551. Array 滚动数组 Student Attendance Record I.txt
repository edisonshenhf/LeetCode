class Solution {
    public boolean checkRecord(String s) {
        int cntA = 0;
        int cntL = 0;
        for(char c:s.toCharArray()){
            if(c=='A') cntA++;
            if(c=='L') cntL++;
            else cntL=0;
            if(cntA>=2||cntL>=3) return false;
        }
        return true;
    }
}
