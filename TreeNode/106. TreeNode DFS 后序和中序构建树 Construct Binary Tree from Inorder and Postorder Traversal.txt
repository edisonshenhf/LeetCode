/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    HashMap<Integer,Integer> map = new HashMap<>();
    public TreeNode buildTree(int[] inorder, int[] postorder) {
        int n = inorder.length;
        for(int i=0;i<n;i++){
            map.put(inorder[i],i);
        }
        return dfs(postorder,inorder,0,n-1,0,n-1);
    }

    TreeNode dfs(int[] postorder,int[] inorder,int pl,int pr,int il,int ir){
        if(pl>pr) return null;
        int t = postorder[pr];
        TreeNode root = new TreeNode(t);
        int k = map.get(t);
        root.left=dfs(postorder,inorder,pl,pl+k-1-il,il,k-1);
        root.right=dfs(postorder,inorder,pl+k-1-il+1,pr-1,k+1,ir);
        return root;

    }
}
