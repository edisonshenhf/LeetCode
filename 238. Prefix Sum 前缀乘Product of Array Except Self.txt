//对于中间的i 从前往后到i-1乘一遍 从后往前到i+1乘一遍

class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] res = new int[n];
        Arrays.fill(res,1);
        for(int i =1;i<n;i++){
            res[i]=res[i-1]*nums[i-1];
        }
        for(int i=n-1,s=1;i>=0;i--){
            res[i]*=s;
            s*=nums[i];
        }
        return res;
    }
}
