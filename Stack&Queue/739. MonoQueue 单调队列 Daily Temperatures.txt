class Solution {
    public int[] dailyTemperatures(int[] temperatures) {
        int n = temperatures.length;
        int[] res = new int[n];
        for(int i=n-1;i>=0;i--){
            for(int j=i+1;j<=n-1;j+=res[j]){
                if(temperatures[j]>temperatures[i]){
                    res[i]=j-i;
                    break;
                }else if(res[j]==0){
                    res[i]=0;
                    break;
                }
            }
        }
        return res;
    }
}
