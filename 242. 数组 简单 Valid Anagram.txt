class Solution {
    public boolean isAnagram(String s, String t) {
        int[] words = new int[26];
        for(int i=0;i<s.length();i++){
            words[s.charAt(i)-'a']++;
        }
        for(int i=0;i<t.length();i++){
            words[t.charAt(i)-'a']--;
        }
        for(int num:words){
            if(num!=0) return false;
        }
        return true;
    }
}
