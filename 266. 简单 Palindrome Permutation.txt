class Solution {
    public boolean canPermutePalindrome(String s) {
        int[] nums = new int[26];
        for(char c:s.toCharArray()){
            nums[c-'a']++;
        }
        int res = 0;
        for(int num:nums){
            if(num%2!=0) res++;
            if(res>=2) return false;
        }
        return true;
    }
}
