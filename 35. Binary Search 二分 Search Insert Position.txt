class Solution {
    public int searchInsert(int[] nums, int target) {
        int l=0;
        int r=nums.length-1;
        while(l<r){
            int mid = l+r>>1;
            if(nums[mid]>=target) r=mid;
            else l=mid+1;
        }
        if(l==nums.length-1&&target>nums[nums.length-1]) return l+1;
        else return l;
    }
}
