class Solution {
    public String intToRoman(int num) {
        int[] nums = {1,4,5,9,10,40,50,90,100,400,500,900,1000};
        String[] chars ={"I","IV","V","IX","X","XL","L","XC","C","CD","D","CM","M"};
        StringBuilder sb = new StringBuilder();
        for(int i=12;i>=0;i--){
            while(num-nums[i]>=0){
                num-=nums[i];
                sb.append(chars[i]);
            }
        }
        return sb.toString();
    }
}
