class Solution {
    public void wiggleSort(int[] nums) {
        int n = nums.length;
        for(int i=0;i<n-1;i++){
            if(i%2==0&&nums[i]>nums[i+1]||i%2==1&&nums[i]<nums[i+1]){
                swap(nums,i,i+1);
            }
        }
    }
    void swap(int[] nums,int l,int r){
        int temp = nums[l];
        nums[l]=nums[r];
        nums[r]=temp;
    }
}
