class Solution {
    public int lengthOfLongestSubstring(String s) {
        int[] nums = new int[128];
        int res = 0;
        for(int i=0,j=0;i<s.length();i++){
                nums[s.charAt(i)]++;
                while(nums[s.charAt(i)]>1){
                    nums[s.charAt(j)]--;
                    j++;
                }
                res= Math.max(res,i-j+1);
        }
        return res;
    }
}
