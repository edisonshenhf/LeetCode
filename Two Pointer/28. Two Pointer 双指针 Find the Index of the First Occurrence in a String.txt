class Solution {
    public int strStr(String haystack, String needle) {
        int n = haystack.length();
        int m = needle.length();
        for(int i=0;i<n;i++){
            for(int j=0;j<=m;j++){
                if(j==m) return i;
                int t = i+j;
                if(t>=n) return -1;
                if(haystack.charAt(t)!=needle.charAt(j)) break;
            }
        }
        return -1;
    }
}
