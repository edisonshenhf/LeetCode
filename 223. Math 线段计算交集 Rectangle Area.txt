//本质上是计算一维线段的交集
//a----b
//   c-----d
//所有线段交集的情况都可以用Math.min(b,d)-Math.max(a,c)来表示
//a----b
//        c----d
//如果没有交集则可以通过Math.max(0,Math.min(b,d)-Math.max(a,c))来判断;
//这样就涵盖了所有情况
//本题就是所有线段投射到x轴与y轴的情况

class Solution {
    public int computeArea(int ax1, int ay1, int ax2, int ay2, int bx1, int by1, int bx2, int by2) {
        return (ax2-ax1)*(ay2-ay1)+(bx2-bx1)*(by2-by1)-Math.max(0,Math.min(ax2,bx2)-Math.max(ax1,bx1))*Math.max(0,Math.min(ay2,by2)-Math.max(ay1,by1));
    }
}
