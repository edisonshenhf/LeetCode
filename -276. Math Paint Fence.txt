class Solution {
    public int numWays(int n, int k) {
        if(n==1) return k;
        if(n==2) return k*k;
        int[] res = new int[n+1];
        res[1]=k;
        res[2]=k*k;
        for(int i=3;i<=n;i++){
            res[i]=(k-1)*(res[i-1]+res[i-2]);
        }
        return res[n];
    }
}
