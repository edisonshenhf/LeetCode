class Solution {
    public int coinChange(int[] coins, int amount) {
        int n = coins.length;
        int[] f = new int[amount+1];
        for(int i=1;i<=amount;i++) f[i]=amount+1;
        for(int i=0;i<n;i++){
            for(int j=coins[i];j<=amount;j++){
                f[j]=Math.min(f[j],f[j-coins[i]]+1);
            }
        }
        return f[amount]>amount?-1:f[amount];
    }
}
