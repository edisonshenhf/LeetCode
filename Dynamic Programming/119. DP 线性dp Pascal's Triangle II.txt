class Solution {
    public List<Integer> getRow(int rowIndex) {
        List<Integer> res = new ArrayList<>();
        int[][] f = new int[rowIndex+1][rowIndex+1];
        for(int i=0;i<=rowIndex;i++){
            for(int j=0;j<=i;j++){
                f[i][0]=1;
                if(i>0&&j>0) f[i][j]=f[i-1][j]+f[i-1][j-1];
                f[i][i]=1;
                if(i==rowIndex) res.add(f[i][j]);
            }
        }
        return res;
    }
}
