class Solution {
    public int combinationSum4(int[] nums, int target) {
        int[] f = new int[target+1];
        f[0]=1;
        for(int i=1;i<=target;i++){
            for(int num:nums){
                if(i>=num){
                    f[i]+=f[i-num];
                }
            }
        }
        return f[target];
    }
}
