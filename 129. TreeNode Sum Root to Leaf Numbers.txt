/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int res;
    public int sumNumbers(TreeNode root) {
        res=0;
        dfs(root,0);
        return res;
    }

    void dfs(TreeNode root,Integer num){
        if(root==null) return;
        num=num*10+root.val;
        if(root.left==null&&root.right==null){
            res+=num;
            return;
        }
        dfs(root.left,num);
        dfs(root.right,num);

    }
}
