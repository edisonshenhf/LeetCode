class Solution {
    public int findMinArrowShots(int[][] points) {
        if(points.length==0) return 0;
        Arrays.sort(points,(a,b)->{
            return a[1]<=b[1]?-1:1;
        });
        int r = points[0][1];
        int res = 1;
        for(int i=1;i<points.length;i++){
            if(points[i][0]>r){
                res++;
                r=points[i][1];
            }
        }
        return res;
    }
}
