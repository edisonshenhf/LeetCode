class Solution {
    public int findJudge(int n, int[][] trust) {
        int[][] nums = new int[n+1][2];
        for(int i=0;i<trust.length;i++){
            nums[trust[i][0]][0]=1;
            nums[trust[i][1]][1]=nums[trust[i][1]][1]+1;
        }
        for(int i=1;i<=n;i++){
            if(nums[i][0]==0&&nums[i][1]==n-1) return i;
        }
        return -1;
    }
}
