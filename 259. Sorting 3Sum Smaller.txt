class Solution {
    public int threeSumSmaller(int[] nums, int target) {
        Arrays.sort(nums);
        int res = 0;
        for(int i=0;i<nums.length-2;i++){
            int j=i+1;
            int k = nums.length-1;
            while(j<k){
                if(nums[j]+nums[k]<target-nums[i]){
                    res+=k-j;
                    j++;
                }
                if(nums[j]+nums[k]>=target-nums[i]) k--;
            }
        }
        return res;
    }
}
