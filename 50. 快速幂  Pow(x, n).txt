class Solution {
    public double myPow(double x, int n) {
        long num = n;
        double t =qmi(x,Math.abs(num));
        if(n<0) return 1/t;
        return t;
    }

    double qmi(double a,long k){
        double res=1;
        while(k>0){
            if((k&1)==1) res*=a;
            k>>=1;
            a*=a;
        }
        return res;
    }
}
